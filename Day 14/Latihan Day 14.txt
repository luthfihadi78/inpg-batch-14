2. function reverseString (text) {
  var arr = []
  for(i=text.length; i>=0; i--){
    arr.push(text[i]);
  }
  return arr.join('')
}

// TEST CASES

function sortByCharacter(str) {
	
	var tampungWords = [];
	for (var i = 0; i < str.length; i++) {
		for (var j = 0; j < str[i].length; j++) {
			if (str[i] !== str[j]) {
				tampungWords.push(str[i][j]);
			} else {
				tampungWords.push(str[i]);
			}
		}
	}
	
	var isWords = 0;
	for (var x = 0; x < tampungWords.length; x++) {
		for (var y = 0; y < tampungWords.length; y++) {
			if (tampungWords[y] > tampungWords[y + 1]) {
				isWords = tampungWords[y];
				tampungWords[y] = tampungWords[y + 1];
				tampungWords[y + 1] = isWords;
			}
		}
	}
	var hasilWords = '';
	for (var k = 0; k < tampungWords.length; k++) {
		hasilWords += tampungWords[k];
	}
	return hasilWords;
}

// TEST CASES
console.log(sortByCharacter('hello')); // 'ehllo'
console.log(sortByCharacter('truncate')); // 'acenrttu'
console.log(sortByCharacter('developer')); // 'deeeloprv'
console.log(sortByCharacter('software')); // 'aeforstw'
console.log(sortByCharacter('aegis')); // 'aegis'

function isArithmeticProgression (numbers) {
 for (var i=0; i<numbers.length; i++){
    return numbers[i+1] - numbers[i] === numbers[numbers.length-1] -  numbers[numbers.length-2];
  }
}


// TEST CASES

console.log(isArithmeticProgression([2, 4, 6, 12, 24])); // false

console.log(isArithmeticProgression([2, 6, 18, 54])); // false
